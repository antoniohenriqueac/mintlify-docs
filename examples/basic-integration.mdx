// Cliente ConnectVets Notes
class ConnectVetsClient {
    constructor(config) {
        this.config = config;
        this.currentNoteId = null;
        this.pollingInterval = null;
    }

    // Fazer requisi√ß√£o com retry autom√°tico
    async request(endpoint, options = {}) {
        const url = `${this.config.baseUrl}${endpoint}`;
        
        const requestOptions = {
            ...options,
            headers: {
                'X-API-KEY': this.config.apiKey,
                ...options.headers
            }
        };

        let lastError;
        
        for (let attempt = 1; attempt <= this.config.maxRetries; attempt++) {
            try {
                console.log(`üîÑ Tentativa ${attempt}/${this.config.maxRetries}: ${endpoint}`);
                
                const response = await fetch(url, requestOptions);
                
                if (response.ok) {
                    return await response.json();
                }
                
                // Tentar parsear erro
                const errorData = await response.json().catch(() => ({}));
                throw new ConnectVetsError(response.status, errorData);
                
            } catch (error) {
                lastError = error;
                
                // N√£o retry em erros de cliente (4xx)
                if (error.status && error.status >= 400 && error.status < 500) {
                    break;
                }
                
                // √öltima tentativa
                if (attempt === this.config.maxRetries) {
                    break;
                }
                
                // Aguardar antes de tentar novamente
                const delay = this.config.retryDelay * attempt;
                console.log(`‚è≥ Aguardando ${delay}ms antes da pr√≥xima tentativa...`);
                await this.sleep(delay);
            }
        }
        
        throw lastError;
    }

    // Criar nova nota
    async createNote(audioFile, metadata) {
        const formData = new FormData();
        formData.append('audio', audioFile);
        formData.append('metadata', JSON.stringify(metadata));

        return this.request('/notes', {
            method: 'POST',
            body: formData,
            headers: {} // N√£o definir Content-Type para FormData
        });
    }

    // Obter status da nota
    async getNote(noteId) {
        return this.request(`/notes/${noteId}`);
    }

    // Iniciar monitoramento de status
    startPolling(noteId, onStatusUpdate, onComplete, onError) {
        this.currentNoteId = noteId;
        let attempts = 0;

        const poll = async () => {
            try {
                attempts++;
                console.log(`üìä Verificando status (tentativa ${attempts})...`);
                
                const note = await this.getNote(noteId);
                
                // Atualizar interface
                onStatusUpdate(note.status, attempts);
                
                // Verificar se processamento terminou
                if (note.status === 'completed') {
                    this.stopPolling();
                    onComplete(note);
                } else if (note.status === 'failed') {
                    this.stopPolling();
                    onError(new Error(note.error_message || 'Processamento falhou'));
                } else if (attempts >= this.config.polling.maxAttempts) {
                    this.stopPolling();
                    onError(new Error('Timeout: Processamento demorou mais que esperado'));
                }
                
            } catch (error) {
                console.error('‚ùå Erro ao verificar status:', error);
                this.stopPolling();
                onError(error);
            }
        };

        // Primeira verifica√ß√£o imediata
        poll();
        
        // Agendar verifica√ß√µes peri√≥dicas
        this.pollingInterval = setInterval(poll, this.config.polling.intervalMs);
    }

    // Parar monitoramento
    stopPolling() {
        if (this.pollingInterval) {
            clearInterval(this.pollingInterval);
            this.pollingInterval = null;
        }
    }

    // Helper: sleep
    sleep(ms) {
        return new Promise(resolve => setTimeout(resolve, ms));
    }
}

    constructor(status, errorData) {
        const message = errorData.error?.message || errorData.message || 'Erro desconhecido';
        super(message);
        
        this.name = 'ConnectVetsError';
        this.status = status;
        this.code = errorData.error?.code || 'unknown_error';

    }
}
